Control Structures Cheat Sheet for Ruby and Javascript

Branching

Ruby Javascript
if

Syntax

if condition 
  ...code...
end

Example

n = 0
if n == 5
  n = 10
end
print n # prints 0

Syntax

if (condition) {
  ...code...
}

Example

var n = 0;
if (n == 5) {
  n = 10;
}
console.log(n); // prints 0
unless

Syntax

unless condition 
  ...code...
end

Example

n = 0
unless n == 5
  n = 10
end
print n # prints 10

Syntax

There is no unless statement in Javascript. You can negate the condition of an if statement to acheive the same result. 
if (!condition) {
  ...code...
}

Example

var n = 0;
if (!(n == 5)) {
  n = 10;
}
console.log(n); // prints 10
if, else if, else

Syntax

 if condition
   ...code block...
 elsif condition2
   ...code block...
 elsif condition3
   ...code block...
 else
   ...code block...
 end

Example

n = 0
if n == 5
  n = 10
elsif n == 4
  n = 9
elsif n == 3
  n = 8
else
  n = 7
end
print n # prints 7

Syntax

if (condition) {
  ...code block...
} else if (condition2) {
  ...code block...
} else if (condition3) {
  ...code block...
} else {
  ...code block...
}

Example

var n = 0;
if (n == 5) {
  n = 10;
} else if (n == 4) {
  n = 9;
} else if (n == 3) {
  n = 8;
} else {
  n = 7;
}
console.log(n); // prints 7

Syntax

result = condition ? expression-if-true : expression-if-false

Example

pet_name = is_cat ? "Twinkle" : "Bandit"

Syntax

result = (condition) ? expression-if-true : expression-if-false;

Example

petName = (isCat) ? "Twinkle" : "Bandit";
case

Syntax

case variable
when value then ...code...    
when value then ...code... 
when value then ...code... 
end

Example

n = 5
case n
when 0 then n = 1    
when 2 then n = 3 
when 5 then n = 6
else n = 8
end
print n # prints 6

Syntax

switch (variable) {
case value: 
  ...code...
  break;
case value:
  ...code... 
  break;
case value: 
  ...code...
  break;
default:
  ...code...
}

Example

var n = 5;
switch (n) {
case 0: 
  n = 1;
  break;
case 2:
  n = 3;
  break;
case 5: 
  n = 6;
  break;
default:
  n = 8;
}
console.log(n) // prints 6

Looping

Ruby Javascript
while

Syntax

while condition_is_true
  # do something
end

Example

i = 1
while i < 5
  puts "#{i} is less than 5!"
  i += 1
end
puts "Done!"

# 1 is less than 5!
# 2 is less than 5!
# 3 is less than 5!
# 4 is less than 5!
# Done!

Syntax

while (condition) {
  code block to be executed
}

Example


var i = 1;
while (i < 5) {
    text += "The number is " + i;
    i++;                     
}
                        
// The number is 1
// The number is 2
// The number is 3
// The number is 4
for

Syntax

for iterator in iterable_object
  ...code...
end

Example

array = [1,2,3,4]  
for item in array
  print item
end
# prints 1, 2, 3, 4

Syntax

for (init-expression; condition; increment-expression) {
  ...code...
}

Example

var array = [1,2,3,4];
for (var i = 0; i < array.length; i++) {
  var item = array[i];
  console.log(item);
}
// prints 1, 2, 3, 4
until

Syntax

until condition_is_false
  ...code...
end

Example

counter = 3
until counter == 0
  puts counter
  counter -= 1
end
puts "Blast off!"

# 3
# 2
# 1
# Blast off!

Syntax

There is no until statement in Javascript. You can negate the condition of an while statement to acheive the same result. 
while (!condition) {
    ...code...
}

Example

var counter = 3;
while (!counter == 0)
  console.log(counter);
  counter -= 1;
}
console.log("Blast off!");

// 3
// 2
// 1
// Blast off!
do while

Syntax

There is no do while statement in Ruby. Yukihiro Matsumoto, the creator of Ruby, recommended this format for a "do at least once" loop. 
loop do 
  ...code...
  break unless/if 
end

Example

n = 0
loop do
   n += 1
   puts n
   break unless n % 6 != 0
end condition
# prints 1, 2, 3, 4, 5, 6

Syntax

do {
    ...code...
} while (condition)

Example

var n = 0;
do {
   n += 1
   console.log(n);
} while (n % 6 != 0)
// prints 1, 2, 3, 4, 5, 6
each

Syntax

array.each do |item| 
  ...code...
end

Example

array = [1,2,3]
array.each do |x| 
   print x
end
# prints 1, 2, 3

Syntax

    
array.forEach(function(item, index, array) {
  ..code...
})

Example

array = [1,2,3];
array.forEach(function(x) {
    console.log(x);
}) 
// prints 1, 2, 3
